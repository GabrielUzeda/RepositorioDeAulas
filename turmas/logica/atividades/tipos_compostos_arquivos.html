    Imagine que o arquivo log.txt não existe. Qual será o conteúdo final de log.txt depois que este script for executado DUAS VEZES seguidas?

    
import os

NOME_ARQUIVO = 'log.txt'

if not os.path.exists(NOME_ARQUIVO):
    with open(NOME_ARQUIVO, 'w') as f:
        f.write('Registro iniciado.\n')
else:
    with open(NOME_ARQUIVO, 'a') as f:
        f.write('Nova entrada no registro.\n')
